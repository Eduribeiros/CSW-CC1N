import numpy as np
    import pandas as pd
    
    def remove_outliers(dataframe, column_name):
        mean = dataframe[column_name].mean()
        std_dev = dataframe[column_name].std()
        
        z_scores = (dataframe[column_name] - mean) / std_dev
        
        dataframe_cleaned = dataframe[np.abs(z_scores) <= 3]
        
        return dataframe_cleaned
    
    df = pd.DataFrame({
        'coluna_de_interesse': [10, 12, 14, 15, 100, 13, 9, 11, 7, 15]
    })
    
    df_limpo = remove_outliers(df, 'coluna_de_interesse')
    
    print(df_limpo)
